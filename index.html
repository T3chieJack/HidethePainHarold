Image Replacer Tool

Generate the files needed to create a browser extension for replacing images dynamically!

How It Works

This tool lets you generate JavaScript, CSS, and a manifest file to create a Chrome extension. Follow the instructions below.

Code Preview

JavaScript Code

function replaceImages() {
  const allImages = document.getElementsByTagName("img");
  for (let i = 0; i < allImages.length; i++) {
    const imageNumber = (i % 10) + 1;
    allImages[i].src = \`https://t3chiejack.github.io/Images/\${imageNumber}.png\`;
  }
}

const observer = new MutationObserver(replaceImages);
observer.observe(document.body, { childList: true, subtree: true });
replaceImages();

Manifest.json

{
  "manifest_version": 3,
  "name": "Image Replacer Tool",
  "version": "1.0",
  "content_scripts": [
    {
      "matches": [""],
      "js": ["image-replacer.js"]
    }
  ]
}
